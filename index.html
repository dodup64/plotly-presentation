Data generation



import plotly.plotly as py
import plotly.tools as tls
from plotly.graph_objs import *
from numpy import pi, cos, exp

# Define the function to be plotted
def fxy(x, y, A=1):
    return A*(cos(pi*x*y))**2 * exp(-(x**2+y**2)/2.)

# Choose length of square domain, make row and column vectors
L = 4
x = y = np.arange(-L/2., L/2., 0.1)  # use a mesh spacing of 0.1
yt = y[:, np.newaxis]  # (!) make column vector

# Get surface coordinates!
z = fxy(x, yt)

Plotting


data = Data([Surface(z=z, x=x, y=y)])
layout = Layout(title='$f(x,y) = A \cos(\pi x y) e^{-(x^2+y^2)/2}$')
fig = Figure(data=data, layout=layout)
py.iplot(fig, filename='s8_surface')

From a R ggplot2 plot


library(plotly)
py <- plotly(plotly_user, plotly_password)

library(ggplot2)
library(gridExtra)
set.seed(10005)
 
xvar <- c(rnorm(1500, mean = -1), rnorm(1500, mean = 1.5))
yvar <- c(rnorm(1500, mean = 1), rnorm(1500, mean = 1.5))
zvar <- as.factor(c(rep(1, 1500), rep(2, 1500)))
xy <- data.frame(xvar, yvar, zvar)
plot<-ggplot(xy, aes(xvar)) + geom_histogram()
py$ggplotly()  # add this to your ggplot2 script to call plotly
